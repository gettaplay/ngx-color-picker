Angular Color Picker Directive

A Color Picker Directive for Angular with no dependencies.

based on angular2-color-picker by Alberto Pujante

Change cancel button class, in this example we are using a bootstrap button: 

<input [value]="color"
       [style.background]="color"
       [cpCancelButton]="true"
       [cpCancelButtonClass]= "'btn btn-primary btn-xs'"
       [(colorPicker)]="color"/>

Show OK button:

<input [value]="color"
       [style.background]="color"
       [cpOKButton]="true"
       [cpSaveClickOutside]="false"
       [cpCancelButton]="true"
       [cpOKButtonClass]= "'btn btn-primary btn-xs'"
       [(colorPicker)]="color"/>

Change event color:

<input [style.background]="color"
       [colorPicker]="color"
       [cpCmykEnabled]="true"
       (cpCmykColorChange)="cmykValue=$event"
       (colorPickerChange)="cmykColor=onChangeColorCmyk($event);color=$event"/>

<span [style.font-size.px]="100 * cmykColor.c"/>C</span/>
<span [style.font-size.px]="100 * cmykColor.m"/>M</span/>
<span [style.font-size.px]="100 * cmykColor.y"/>Y</span/>
<span [style.font-size.px]="100 * cmykColor.k"/>K</span/>

With preset colors:

<input [style.background]="color"
       [cpPresetColors]="['#fff', '#000', '#2889e9', '#e920e9', '#fff500', 'rgb(236,64,64)']"
       [(colorPicker)]="color"/>

Add and remove preset colors:

<input [style.background]="color"
       [cpAlphaChannel]="'always'"
       [cpOutputFormat]="'rgba'"
       [cpPresetColors]="['#fff', '#2889e9']"
       [cpAddColorButton]="true"
       [(colorPicker)]="color"/>


Toggle status: {{ toggle }}

Use cpToggle with cpIgnoredElements:

<input #ignoredInput
       [style.background]="color"
       [cpIgnoredElements]="[ignoredButton, ignoredInput]"
       [(cpToggle)]="toggle"
       [(colorPicker)]="color"/>

<button #ignoredButton (click)="toggle=!toggle"></button>

Auto positioning:

<input [value]="color"
       [style.background]="color"
       cpPosition="auto"
       [(colorPicker)]="color"/>
